```tsx
'use client';

import { useRef } from 'react';

export type SectionId = 'website' | 'vision' | 'brand' | 'tools';

interface SectionCardProps {
  id: SectionId;
  label: string;
  bg: string;
  textColor: string;
  labelColor?: string;
  onOpen: (id: SectionId, rect: DOMRect) => void;
  cardRef?: React.RefObject<HTMLButtonElement | null>;
}

export default function SectionCard({
  id,
  label,
  bg,
  textColor,
  onOpen,
  cardRef,
}: SectionCardProps) {
  const internalRef = useRef<HTMLButtonElement>(null);
  const ref = cardRef ?? internalRef;

  function handleClick() {
    const el = ref.current;
    if (!el) return;
    onOpen(id, el.getBoundingClientRect());
  }

  return (
    <button
      ref={ref}
      onClick={handleClick}
      style={{ background: bg, color: textColor }}
      className="relative flex items-end justify-start overflow-hidden rounded-sm cursor-pointer transition-transform duration-300 hover:scale-[1.03] border-0 p-0 w-full"
      aria-label={`Open ${label} section`}
    >
      <span
        className="relative z-10 p-5 font-display font-medium text-lg leading-none"
        style={{ color: textColor }}
      >
        {label}
      </span>
    </button>
  );
}
```

- [ ] **Step 4: Create HomeGrid.tsx**

```tsx
'use client';

import { useState, useCallback } from 'react';
import SectionCard, { type SectionId } from './SectionCard';
import ZoomShell from './ZoomShell';
import WebsiteSection from './sections/WebsiteSection';
import VisionSection from './sections/VisionSection';
import BrandSection from './sections/BrandSection';
import ToolsSection from './sections/ToolsSection';

interface CardConfig {
  id: SectionId;
  label: string;
  bg: string;
  textColor: string;
}

const CARDS: CardConfig[] = [
  { id: 'website', label: 'Website', bg: '#1A1813', textColor: '#ffffff' },
  { id: 'vision',  label: 'Vision',  bg: '#BB3308', textColor: '#ffffff' },
  { id: 'brand',   label: 'Brand',   bg: '#14122B', textColor: '#ffffff' },
  { id: 'tools',   label: 'Tools',   bg: '#0e0e0a', textColor: '#51C8FF' },
];

const SECTION_CONTENT: Record<SectionId, React.ReactNode> = {
  website: <WebsiteSection />,
  vision:  <VisionSection />,
  brand:   <BrandSection />,
  tools:   <ToolsSection />,
};

const SECTION_BG: Record<SectionId, string> = {
  website: '#1A1813',
  vision:  '#BB3308',
  brand:   '#14122B',
  tools:   '#0e0e0a',
};

interface FlyingRect {
  top: number;
  left: number;
  width: number;
  height: number;
}

export default function HomeGrid() {
  const [activeSection, setActiveSection] = useState<SectionId | null>(null);
  const [flyingRect, setFlyingRect] = useState<FlyingRect | null>(null);
  const [isZoomed, setIsZoomed] = useState(false);
  const [isClosing, setIsClosing] = useState(false);

  const openSection = useCallback((id: SectionId, rect: DOMRect) => {
    setFlyingRect({ top: rect.top, left: rect.left, width: rect.width, height: rect.height });
    setActiveSection(id);
    // Double rAF: wait for flying screen to render at card position, then trigger expand
    requestAnimationFrame(() => {
      requestAnimationFrame(() => setIsZoomed(true));
    });
  }, []);

  const closeSection = useCallback(() => {
    setIsClosing(true);
    setIsZoomed(false);
    setTimeout(() => {
      setActiveSection(null);
      setFlyingRect(null);
      setIsClosing(false);
    }, 650);
  }, []);

  const flyingStyle: React.CSSProperties = flyingRect
    ? {
        position: 'fixed',
        zIndex: 101,
        overflow: 'hidden',
        top:    isZoomed ? '40px'                   : flyingRect.top,
        left:   isZoomed ? '40px'                   : flyingRect.left,
        width:  isZoomed ? 'calc(100vw - 80px)'     : flyingRect.width,
        height: isZoomed ? 'calc(100vh - 80px)'     : flyingRect.height,
        borderRadius: isZoomed ? '16px' : '4px',
        background: activeSection ? SECTION_BG[activeSection] : 'transparent',
        transition: 'top 0.6s cubic-bezier(0.4,0,0.2,1), left 0.6s cubic-bezier(0.4,0,0.2,1), width 0.6s cubic-bezier(0.4,0,0.2,1), height 0.6s cubic-bezier(0.4,0,0.2,1), border-radius 0.6s cubic-bezier(0.4,0,0.2,1)',
        pointerEvents: isZoomed ? 'all' : 'none',
      }
    : {};

  return (
    <>
      {/* ── Dark overlay ── */}
      <div
        className="fixed inset-0 z-50 bg-[#0e0e0a] transition-opacity duration-500 pointer-events-none"
        style={{ opacity: isZoomed ? 1 : 0 }}
      />

      {/* ── Index view ── */}
      <div
        className="min-h-screen flex flex-col items-center justify-center px-10 py-12 transition-opacity duration-400"
        style={{ opacity: (activeSection && !isClosing) ? 0 : 1 }}
      >
        <p className="font-mono text-[11px] tracking-[0.2em] uppercase text-white/40 mb-12">
          Bhutan Innovation Festival²⁷
        </p>

        <div className="flex flex-wrap gap-5 w-full max-w-[1200px] justify-center">
          {CARDS.map((card) => (
            <SectionCard
              key={card.id}
              id={card.id}
              label={card.label}
              bg={card.bg}
              textColor={card.textColor}
              onOpen={openSection}
            />
          ))}
        </div>
      </div>

      {/* ── Flying screen (animates card → full viewport) ── */}
      {flyingRect && activeSection && (
        <div style={flyingStyle}>
          {isZoomed && (
            <ZoomShell sectionId={activeSection} onClose={closeSection}>
              {SECTION_CONTENT[activeSection]}
            </ZoomShell>
          )}
        </div>
      )}
    </>
  );
}
```

- [ ] **Step 3: Create AgentToggle.tsx**

```tsx
'use client';

import { useState } from 'react';

export default function AgentToggle() {
  const [isAgent, setIsAgent] = useState(false);

  function toggle() {
    const next = !isAgent;
    setIsAgent(next);
    document.documentElement.setAttribute('data-view-mode', next ? 'agent' : 'human');
  }

  return (
    <button
      onClick={toggle}
      className="flex items-center gap-2.5 cursor-pointer border-0 bg-transparent font-mono text-[12px] tracking-[0.08em] transition-colors duration-300"
      style={{ color: isAgent ? 'rgba(255,255,255,0.7)' : 'rgba(255,255,255,0.35)' }}
      aria-label={isAgent ? 'Switch to human view' : 'Switch to agent view'}
    >
      <svg width="16" height="16" viewBox="0 0 16 16" fill="none" style={{ opacity: isAgent ? 0.8 : 0.5 }}>
        <rect x="2" y="2" width="5" height="5" rx="1" stroke="currentColor" strokeWidth="1.2" />
        <rect x="9" y="2" width="5" height="5" rx="1" stroke="currentColor" strokeWidth="1.2" />
        <rect x="2" y="9" width="5" height="5" rx="1" stroke="currentColor" strokeWidth="1.2" />
        <rect x="9" y="9" width="5" height="5" rx="1" stroke="currentColor" strokeWidth="1.2" />
      </svg>
      {isAgent ? 'agent view' : 'agent mode'}
    </button>
  );
}
```
